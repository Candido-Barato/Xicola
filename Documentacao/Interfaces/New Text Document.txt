SELECT Curso.id_Curso, nivel_curso.id_Nivel_Curso, concat(Curso.Curso,' - ',nivel_curso.Nivel_Curso, ' - ', modulo_curso.Modulo_Curso) as 'Descricao Curso',nivel_curso.Organizacao,nivel_curso.`Forma Pagamento`, nivel_curso.Valor FROM `curso`, nivel_curso,modulo_nivel,modulo_curso WHERE Curso.id_Curso=nivel_curso.id_Curso and modulo_nivel.id_Nivel_Curso=nivel_curso.id_Nivel_Curso and modulo_curso.idModulo_Curso=modulo_nivel.id_Modulo
